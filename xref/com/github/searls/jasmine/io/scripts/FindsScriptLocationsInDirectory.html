1   package com.github.searls.jasmine.io.scripts;
2   
3   import java.io.File;
4   import java.util.ArrayList;
5   import java.util.List;
6   
7   import com.github.searls.jasmine.io.ScansDirectory;
8   import com.github.searls.jasmine.model.ScriptSearch;
9   
10  public class FindsScriptLocationsInDirectory {
11  
12    private final ScansDirectory scansDirectory = new ScansDirectory();
13    private final ConvertsFileToUriString convertsFileToUriString = new ConvertsFileToUriString();
14  
15    public List<String> find(ScriptSearch search) {
16      List<String> scriptLocations = new ArrayList<String>();
17      if(search.getDirectory().canRead()) {
18        for(String script : scansDirectory.scan(search.getDirectory(), search.getIncludes(), search.getExcludes())) {
19          scriptLocations.add(convertsFileToUriString.convert(new File(search.getDirectory(),script)));
20        }
21      }
22      return scriptLocations;
23    }
24  }
