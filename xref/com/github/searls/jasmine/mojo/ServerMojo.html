16  @Mojo(name="bdd",requiresDirectInvocation=true)

19    public static final String INSTRUCTION_FORMAT =
20        "\n\n" +
21            "Server started--it's time to spec some JavaScript! You can run your specs as you develop by visiting this URL in a web browser: \n\n" +
22            "  http://localhost:%s"+
23            "\n\n" +
24            "The server will monitor these two directories for scripts that you add, remove, and change:\n\n" +
25            "  source directory: %s\n\n"+
26            "  spec directory: %s"+
27            "\n\n"+
28            "Just leave this process running as you test-drive your code, refreshing your browser window to re-run your specs. You can kill the server with Ctrl-C when you're done.";
29  
30    private final RelativizesFilePaths relativizesFilePaths;
31  
32    public ServerMojo() {
33      this.relativizesFilePaths = new RelativizesFilePaths();
34    }
35  
36    private String buildServerInstructions() throws IOException {
37      return String.format(
38          INSTRUCTION_FORMAT,
39          this.serverPort,
40          this.getRelativePath(this.sources.getDirectory()),
41          this.getRelativePath(this.specs.getDirectory()));
42    }
43  
44    @Override
45    public void run() throws Exception {
46      ServerManager serverManager = new ServerManager(new ResourceHandlerConfigurator(
47          this,
48          this.relativizesFilePaths,
49          this.manualSpecRunnerHtmlFileName,
50          ReporterType.HtmlReporter));
51      serverManager.start(this.serverPort);
52      this.getLog().info(this.buildServerInstructions());
53      serverManager.join();
54    }
55  
56    private String getRelativePath(File absolutePath) throws IOException {
57      return this.relativizesFilePaths.relativize(this.mavenProject.getBasedir(), absolutePath);
58    }
