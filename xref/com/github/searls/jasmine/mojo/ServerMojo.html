17  /**
18   * Execute specs in a web browser. Monitors your sources/specs for changes as you develop.
19   */
20  @Mojo(name="bdd",requiresDirectInvocation=true)
21  public class ServerMojo extends AbstractJasmineMojo {
22  
23    public static final String INSTRUCTION_FORMAT =
24        "\n\n" +
25            "Server started--it's time to spec some JavaScript! You can run your specs as you develop by visiting this URL in a web browser: \n\n" +
26            " %s://localhost:%s"+
27            "\n\n" +
28            "The server will monitor these two directories for scripts that you add, remove, and change:\n\n" +
29            "  source directory: %s\n\n"+
30            "  spec directory: %s"+
31            "\n\n"+
32            "Just leave this process running as you test-drive your code, refreshing your browser window to re-run your specs. You can kill the server with Ctrl-C when you're done.";
33  
34    private final RelativizesFilePaths relativizesFilePaths;
35  
36    public ServerMojo() {
37      this.relativizesFilePaths = new RelativizesFilePaths();
38    }
39  
40    private String buildServerInstructions() throws IOException {
41      return String.format(
42          INSTRUCTION_FORMAT,
43          this.uriScheme,
44          this.serverPort,
45          this.getRelativePath(this.sources.getDirectory()),
46          this.getRelativePath(this.specs.getDirectory()));
47    }
48  
49    @Override
50    public void run() throws Exception {
51      ServerManager serverManager = this.getServerManager();
52  
53      serverManager.start(this.serverPort);
54      this.getLog().info(this.buildServerInstructions());
55      serverManager.join();
56    }
57  
58    private ServerManager getServerManager() throws MojoExecutionException {
59      Log log = this.debug ? this.getLog() : new NullLog();
60  
61      CreatesRunner createsRunner = new CreatesRunner(
62          this,
63          log,
64          this.manualSpecRunnerHtmlFileName,
65          ReporterType.HtmlReporter);
66  
67      ResourceHandlerConfigurator configurator = new ResourceHandlerConfigurator(
68          this,
69          this.relativizesFilePaths,
70          createsRunner);
71  
72      return new ServerManager(new Server(), getConnector(), configurator);
73    }
74  
75    private String getRelativePath(File absolutePath) throws IOException {
76      return this.relativizesFilePaths.relativize(this.mavenProject.getBasedir(), absolutePath);
77    }
