23  /**
24   * Execute specs using Selenium Web Driver. Uses HtmlUnitDriver for head-less execution by default.
25   */
26  @Mojo(name="test",defaultPhase=LifecyclePhase.TEST)
27  public class TestMojo extends AbstractJasmineMojo {
28  
29    private final RelativizesFilePaths relativizesFilePaths;
30  
31    public TestMojo() {
32      this.relativizesFilePaths = new RelativizesFilePaths();
33    }
34  
35    @Override
36    public void run() throws Exception {
37      if(!this.skipTests) {
38        ServerManager serverManager = this.getServerManager();
39        try {
40          int port = serverManager.start();
41          setPortProperty(port);
42          this.getLog().info("Executing Jasmine Specs");
43          JasmineResult result = this.executeSpecs(new URL("http://localhost:"+port));
44          this.logResults(result);
45          this.throwAnySpecFailures(result);
46        } finally {
47          if (!keepServerAlive) {
48            serverManager.stop();
49          }
50        }
51      } else {
52        this.getLog().info("Skipping Jasmine Specs");
53      }
54    }
55  
56    private ServerManager getServerManager() {
57      Log log = this.debug ? this.getLog() : new NullLog();
58  
59      CreatesRunner createsRunner = new CreatesRunner(
60          this,
61          log,
62          this.specRunnerHtmlFileName,
63          ReporterType.JsApiReporter);
64  
65      ResourceHandlerConfigurator configurator = new ResourceHandlerConfigurator(
66          this,
67          this.relativizesFilePaths,
68          createsRunner);
69  
70      return new ServerManager(configurator);
71    }
72  
73    private void setPortProperty(int port) {
74      this.mavenProject.getProperties().setProperty("jasmine.serverPort", String.valueOf(port));
75    }
76    private JasmineResult executeSpecs(URL runner) throws Exception {
77      WebDriver driver = this.createDriver();
78      JasmineResult result = new SpecRunnerExecutor().execute(
79          runner,
80          new File(this.jasmineTargetDir,this.junitXmlReportFileName),
81          driver,
82          this.timeout, this.debug, this.getLog(), this.format);
83      return result;
84    }
85  
86    private WebDriver createDriver() throws Exception {
87      WebDriverFactory factory = new WebDriverFactory();
88      factory.setWebDriverCapabilities(webDriverCapabilities);
89      factory.setWebDriverClassName(webDriverClassName);
90      factory.setDebug(debug);
91      factory.setBrowserVersion(browserVersion);
92      return factory.createWebDriver();
93    }
94  
95    private void logResults(JasmineResult result) {
96      JasmineResultLogger resultLogger = new JasmineResultLogger();
97      resultLogger.setLog(this.getLog());
98      resultLogger.log(result);
99    }
100 
101   private void throwAnySpecFailures(JasmineResult result) throws MojoFailureException {
102     if(this.haltOnFailure && !result.didPass()) {
103       throw new MojoFailureException("There were Jasmine spec failures.");
104     }
