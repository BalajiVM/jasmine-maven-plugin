26  /**
27   * Execute specs using Selenium Web Driver. Uses HtmlUnitDriver for head-less execution by default.
28   */
29  @Mojo(name="test",defaultPhase=LifecyclePhase.TEST,requiresDependencyResolution = ResolutionScope.TEST)
30  public class TestMojo extends AbstractJasmineMojo {
31  
32    private final RelativizesFilePaths relativizesFilePaths;
33  
34    public TestMojo() {
35      this.relativizesFilePaths = new RelativizesFilePaths();
36    }
37  
38    @Override
39    public void execute() throws MojoExecutionException, MojoFailureException {
40      if(!this.isSkipTests()) {
41        super.execute();
42      } else {
43        this.getLog().info("Skipping Jasmine Specs");
44      }
45    }
46  
47    @Override
48    public void run() throws Exception {
49      ServerManager serverManager = this.getServerManager();
50      try {
51        int port = serverManager.start();
52        setPortProperty(port);
53        this.getLog().info("Executing Jasmine Specs");
54        JasmineResult result = this.executeSpecs(new URL(this.uriScheme+"://" + this.serverHostname + ":" + port));
55        this.logResults(result);
56        this.throwAnySpecFailures(result);
57      } finally {
58        if (!keepServerAlive) {
59          serverManager.stop();
60        }
61      }
62    }
63  
64    private ServerManager getServerManager() throws MojoExecutionException {
65      Log log = this.debug ? this.getLog() : new NullLog();
66  
67      CreatesRunner createsRunner = new CreatesRunner(
68          this,
69          log,
70          this.specRunnerHtmlFileName,
71          ReporterType.JsApiReporter);
72  
73      ResourceHandlerConfigurator configurator = new ResourceHandlerConfigurator(
74          this,
75          this.relativizesFilePaths,
76          createsRunner);
77  
78      return new ServerManager(new Server(), getConnector(), configurator);
79    }
80  
81    private void setPortProperty(int port) {
82      this.mavenProject.getProperties().setProperty("jasmine.serverPort", String.valueOf(port));
83    }
84    private JasmineResult executeSpecs(URL runner) throws Exception {
85      WebDriver driver = this.createDriver();
86      JasmineResult result = new SpecRunnerExecutor().execute(
87          runner,
88          new File(this.jasmineTargetDir,this.junitXmlReportFileName),
89          driver,
90          this.timeout, this.debug, this.getLog(), this.format);
91      return result;
92    }
93  
94    private WebDriver createDriver() throws Exception {
95      WebDriverFactory factory = new WebDriverFactory();
96      factory.setWebDriverCapabilities(webDriverCapabilities);
97      factory.setWebDriverClassName(webDriverClassName);
98      factory.setDebug(debug);
99      factory.setBrowserVersion(browserVersion);
100     return factory.createWebDriver();
101   }
102 
103   private void logResults(JasmineResult result) {
104     JasmineResultLogger resultLogger = new JasmineResultLogger();
105     resultLogger.setLog(this.getLog());
106     resultLogger.log(result);
107   }
108 
109   private void throwAnySpecFailures(JasmineResult result) throws MojoFailureException {
110     if(this.haltOnFailure && !result.didPass()) {
111       throw new MojoFailureException("There were Jasmine spec failures.");
112     }
113   }
