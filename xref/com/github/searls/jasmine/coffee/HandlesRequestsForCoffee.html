27    public void handle(Request baseRequest, HttpServletResponse response, Resource resource) throws IOException {
28      baseRequest.setHandled(true);
29      String javascript = null;
30      if (!configuration.isCoffeeScriptCompilationEnabled()) {
31      	// CoffeeScript RequireJS plugin should be used for translation
32      	javascript = IOUtils.toString(resource.getInputStream(), "UTF-8");
33      } else {
34          javascript = compileCoffee(resource);
35      }
36      setHeaders(response, resource, javascript);
37      writeResponse(response, javascript);
38    }
39  
40    private void writeResponse(HttpServletResponse response, String javascript) throws IOException {
41      response.getWriter().write(javascript);
42    }
43  
44    private void setHeaders(HttpServletResponse response, Resource resource, String javascript) {
45      response.setCharacterEncoding("UTF-8");
46      response.setContentType("text/javascript");
47      response.setDateHeader(HttpHeaders.LAST_MODIFIED, resource.lastModified());
48      try {
49        int contentLength = javascript.getBytes("UTF-8").length;
50        response.setHeader(HttpHeaders.CONTENT_LENGTH,Integer.toString(contentLength));
51      } catch (UnsupportedEncodingException e) {
52        throw new RuntimeException(
53          "The JVM does not support the compiler's default encoding.", e);
54      }
55  
56    }
57  
58    private String compileCoffee(Resource resource) {
59      try {
60        return coffeeScript.compile(IOUtils.toString(resource.getInputStream(), "UTF-8"));
61      } catch(Exception e) {
62        return buildsJavaScriptToWriteFailureHtml.build("CoffeeScript Error: failed to compile <code>"+resource.getName()+"</code>. <br/>Error message:<br/><br/><code>"+e.getMessage()+"</code>");
63      }
64    }
