20    private final ThreadLocal<HtmlPage> htmlPage = new ThreadLocal<HtmlPage>() {
21      @Override
22      protected HtmlPage initialValue() {
23        MockWebConnection webConnection = new MockWebConnection();
24        WebClient webClient = new WebClient();
25        webClient.setWebConnection(webConnection);
26        try {
27          HtmlPage page = webClient.getPage(WebClient.URL_ABOUT_BLANK);
28          page.executeJavaScript(ioUtilsWrapper.toString("/vendor/js/coffee-script.js"));
29          return page;
30        } catch (IOException e) {
31          throw new RuntimeException(e);
32        }
33      }
34    };

36    private final IOUtilsWrapper ioUtilsWrapper = new IOUtilsWrapper();
37  
38    public String compile(String coffee) throws IOException {
39      String escapedCoffee = StringEscapeUtils.escapeEcmaScript(coffee);
40      return cache.containsKey(escapedCoffee) ? cache.get(escapedCoffee) : compileAndCache(escapedCoffee);
41    }
42  
43    private String compileAndCache(String input) {
44      ScriptResult scriptResult = htmlPage.get().executeJavaScript(String.format("CoffeeScript.compile(\"%s\");", input));
45      String result = (String) scriptResult.getJavaScriptResult();
46      cache.put(input,result);
47      return result;
48    }
