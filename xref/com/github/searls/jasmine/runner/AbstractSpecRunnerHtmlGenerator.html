1   package com.github.searls.jasmine.runner;

3   import java.io.IOException;
4   import java.util.List;
5   import java.util.Set;
6   
7   import org.codehaus.plexus.util.StringUtils;
8   import org.stringtemplate.v4.ST;
9   
10  import com.github.searls.jasmine.format.FormatsScriptTags;
11  
12  public abstract class AbstractSpecRunnerHtmlGenerator {
13    private static final String SOURCE_ENCODING = "sourceEncoding";
14    private static final String CSS_DEPENDENCIES_TEMPLATE_ATTR_NAME = "cssDependencies";
15    private static final String JAVASCRIPT_DEPENDENCIES_TEMPLATE_ATTR_NAME = "javascriptDependencies";
16    protected static final String SOURCES_TEMPLATE_ATTR_NAME = "sources";
17    protected static final String REPORTER_ATTR_NAME = "reporter";
18    private final HtmlGeneratorConfiguration configuration;
19    private final FormatsScriptTags formatsScriptTags = new FormatsScriptTags();
20  
21    protected AbstractSpecRunnerHtmlGenerator(HtmlGeneratorConfiguration configuration) {
22      this.configuration = configuration;
23    }
24  
25    protected void setEncoding(HtmlGeneratorConfiguration htmlGeneratorConfiguration, ST template) {
26      template.add(SOURCE_ENCODING, StringUtils.isNotBlank(htmlGeneratorConfiguration.getSourceEncoding()) ? htmlGeneratorConfiguration.getSourceEncoding() : SpecRunnerHtmlGenerator.DEFAULT_SOURCE_ENCODING);
27    }
28  
29    protected ST resolveHtmlTemplate() throws IOException {
30      String htmlTemplate = configuration.getRunnerTemplate();
31      return new ST(htmlTemplate,'$','$');
32    }
33  
34    protected void includeJavaScriptDependencies(List<String> dependencies, ST template) throws IOException {
35      StringBuilder js = new StringBuilder();
36      for (String jsFile : dependencies) {
37        if(jsFile != null) {
38          js.append("<script type=\"text/javascript\">").append(configuration.IOtoString(jsFile)).append("</script>");
39        }
40      }
41      template.add(JAVASCRIPT_DEPENDENCIES_TEMPLATE_ATTR_NAME, js.toString());
42    }
43  
44    protected void applyCssToTemplate(List<String> styles, ST template) throws IOException {
45      StringBuilder css = new StringBuilder();
46      for (String cssFile : styles) {
47        css.append("<style type=\"text/css\">").append(configuration.IOtoString(cssFile)).append("</style>");
48      }
49      template.add(CSS_DEPENDENCIES_TEMPLATE_ATTR_NAME, css.toString());
50    }
51  
52    public HtmlGeneratorConfiguration getConfiguration() {
53      return configuration;
54    }
55  
56    protected void applyScriptTagsToTemplate(String sourcesTemplateAttrName, Set<String> scripts, ST template) throws IOException {
57      template.add(sourcesTemplateAttrName, formatsScriptTags.format(scripts));
58    }
59  }
