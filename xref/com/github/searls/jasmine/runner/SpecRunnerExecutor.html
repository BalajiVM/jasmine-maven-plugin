24  	public JasmineResult execute(URL runnerUrl, File junitXmlReport, WebDriver driver, int timeout, boolean debug, Log log, String format) {
25  		try {
26  			if (!(driver instanceof JavascriptExecutor)) {
27  				throw new RuntimeException("The provided web driver can't execute JavaScript: " + driver.getClass());
28  			}
29  			JavascriptExecutor executor = (JavascriptExecutor) driver;
30  			driver.get(runnerUrl.toString());
31  			this.waitForRunnerToFinish(driver, timeout, debug, log);
32  			JasmineResult jasmineResult = new JasmineResult();
33  			jasmineResult.setDetails(this.buildReport(executor,format));
34  			FileUtils.writeStringToFile(junitXmlReport, this.buildJunitXmlReport(executor,debug), "UTF-8");
35  			driver.quit();
36  
37  			return jasmineResult;
38  		} catch (Exception e) {
39  			throw new RuntimeException(e);
40  		}
41  	}

43  	private String buildReport(JavascriptExecutor driver, String format) throws IOException {
44  		String script =
45  				this.ioUtilsWrapper.toString(BUILD_REPORT_JS) +
46  				"return jasmineMavenPlugin.printReport(window.reporter,{format:'"+format+"'});";
47  		Object report = driver.executeScript(script);
48  		return report.toString();
49  	}

51  	private String buildJunitXmlReport(JavascriptExecutor driver, boolean debug) throws IOException {
52  		Object junitReport = driver.executeScript(
53  				this.ioUtilsWrapper.toString(CREATE_JUNIT_XML) +
54  				"return junitXmlReporter.report(window.reporter,"+debug+");");
55  		return junitReport.toString();
56  	}

58  	private void waitForRunnerToFinish(final WebDriver driver, int timeout, boolean debug, Log log) throws InterruptedException {
59  		final JavascriptExecutor executor = (JavascriptExecutor) driver;
60  		new WebDriverWait(driver, timeout, 1000).until(new Predicate<WebDriver>() {
61  			@Override
62  			public boolean apply(WebDriver input) {
63  				return SpecRunnerExecutor.this.executionFinished(executor);
64  			}
65  		});
66  
67  		if (!this.executionFinished(executor)) {
68  			this.handleTimeout(timeout, debug, log);
69  		}
70  	}

72  	private void handleTimeout(int timeout, boolean debug, Log log) {
73  		log.warn("Attempted to wait for your specs to finish processing over the course of " +
74  				timeout +
75  				" seconds, but it still appears to be running.");
76  		if(debug) {
77  			log.warn("Debug mode: will attempt to parse the incomplete spec runner results");
78  		} else {
79  			throw new IllegalStateException("Timeout occurred. Aborting execution of specs. (Try configuring 'debug' to 'true' for more details.)");
80  		}
81  	}

83  	private Boolean executionFinished(JavascriptExecutor driver) {
84  		return (Boolean) driver.executeScript("return (window.reporter === undefined) ? false : window.reporter.finished");
85  	}
